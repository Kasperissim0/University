if (n <= 0) return false; if (start == target) return true;

  int saved = 0, head = 0, tail = 0, size = n, *toVisit = new int[n]; bool *visited = new bool[n]; for (int i = 0; i < n; ++i) visited[i] = false;
  const auto cleanUp = [&]() { delete[] visited, delete[] toVisit; };
  const auto empty = [&]() { return saved == 0; };
  const auto get = [&]() {
    // no need to check for !empty() since only called inside while loop
    return --saved, toVisit[tail++ % size];
  };
  const auto add = [&](const int &val) {
    if (++saved > size) {
      int oldSize = size, *newStorage = new int[size *= (size + 1)]; // agressive scaling to avoid reallocations
      for (int i = 0; i < saved; ++i) newStorage[i] = toVisit[(tail + i) % oldSize];
      tail = 0; head = saved, delete[] toVisit; toVisit = newStorage; 
    }
    return toVisit[head++ % size] = val;
  }; visited[add(start)] = true;

  while (!empty()) {
    const auto *edge = adjList[get()].get_head();
    while (edge) {
      auto nodeValue = edge->data;
      if (!visited[nodeValue]) {
        if (nodeValue == target) return cleanUp(), true;
        visited[add(nodeValue)] = true;
      }
      edge = edge->next;
    }
  }
  return cleanUp(), false;
